db_contacts=# create table contacts (
c_id SERIAL PRIMARY KEY,
phone varchar(15) NOT NULL,
email varchar(30) NOT NULL,
summary varchar(30) NOT NULL,
nickname varchar(15) NOT NULL,
address varchar(30) NOT NULL,
cat_id INTEGER NOT NULL
);

db_contacts=# create table categories (
cat_id SERIAL PRIMARY KEY,
name varchar(15) NOT NULL);

//Agregamos la restriccion de la fk en la tabla contacts
db_contacts=# alter table contacts
add constraint fk_contacts_categories
foreign key (cat_id)
references categories (cat_id);
